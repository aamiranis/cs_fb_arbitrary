0028 function [A]=sgwt_randmat(N,thresh)
0029   assert(thresh<=1 && thresh>=0);
0030   A=rand(N)>1-thresh;
0031   B=triu(A);
0032   A=B+B';
0033   for i=1:size(A,1)
0034     A(i,i)=0;
0035   end
0036   A=sparse(A);
